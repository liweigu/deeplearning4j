	1) cmake
	wget https://cmake.org/files/v3.10/cmake-3.10.2.tar.gz
	tar -xf cmake-3.10.2.tar.gz && cd cmake-3.10.2
	./bootstrap
	make -j8
	#make
	make install
	cmake --version
	2) OpenBlas
	git clone git://github.com/xianyi/OpenBLAS
	cd OpenBLAS
	# make FC=gfortran
	make
	make PREFIX=/usr/local/ install
	3) gfortran
	http://sxyqhyt.iteye.com/blog/2322872
	wget http://gfortran.meteodat.ch/download/x86_64/nightlies/gcc-trunk.tar.xz
	tar -xf gcc-trunk.tar
	mv gcc-trunk  /opt
	export  LD_LIBRARY_PATH="/opt/gcc-trunk/lib64:$LD_LIBRARY_PATH"
	export PATH=/opt/gcc-trunk/bin:$PATH
	gfortran --version
